  for i in range(n):
    hue = i * 360 / n      # n = species count in group, sorted by canonical_name asc
    saturation = 0.7
    lightness = 0.5
    rgb = colorsys.hls_to_rgb(hue/360, lightness, saturation)
    hex = '#%02x%02x%02x' % tuple(round(c * 255) for c in rgb)
