```python
def _slugify(value: str) -> str:
    """Convert a human name or place name to a URL-safe ASCII slug."""
    value = unicodedata.normalize('NFKD', value)
    value = value.encode('ascii', 'ignore').decode('ascii')
    value = value.lower()
    value = re.sub(r'[\s_.,]+', '-', value)
    value = re.sub(r'[^a-z0-9-]', '', value)
    value = re.sub(r'-+', '-', value)
    return value.strip('-') or 'unknown'
```

    ```python
    def _jsonify_rows(rows: list[dict]) -> list[dict]:
        """Convert DuckDB rows to JSON-safe dicts.

        - datetime.date → ISO string
        - month_histogram (list of int) passes through
        - all other values pass through unchanged
        """
        out = []
        for r in rows:
            j = {}
            for k, v in r.items():
                if hasattr(v, 'isoformat'):  # datetime.date / datetime.datetime
                    j[k] = v.isoformat()
                else:
                    j[k] = v
            out.append(j)
        return out
    ```
