    ```python
    def test_idempotency_two_runs(fixture_con, export_dir, monkeypatch):
        """Success criterion 4 / Pitfall #6: two consecutive runs produce byte-identical artifacts."""
        import hashlib
        import time
        import species_export
        monkeypatch.setattr(species_export, 'ASSETS_DIR', export_dir)

        artifacts = ['species.parquet', 'species.json', 'seasonality.json']

        species_export.export_species_parquet(fixture_con)
        first = {
            name: hashlib.sha256((export_dir / name).read_bytes()).hexdigest()
            for name in artifacts
        }

        time.sleep(1.5)  # Make any time-dependent non-determinism observable

        species_export.export_species_parquet(fixture_con)
        second = {
            name: hashlib.sha256((export_dir / name).read_bytes()).hexdigest()
            for name in artifacts
        }

        for name in artifacts:
            assert first[name] == second[name], (
                f"{name} differs between runs (first={first[name]}, second={second[name]})"
            )
    ```

    ```python
        from feeds import _slugify

        import species_maps

        species_maps.generate_species_maps(fixture_con)

        rows = duckdb.execute(f"""
            SELECT scientificName, canonical_name, slug, occurrence_count
            FROM read_parquet('{export_dir / "species.parquet"}')
        """).fetchall()

        maps_dir = export_dir / 'species-maps'
        for sci, canon, slug, count in rows:
    ```
