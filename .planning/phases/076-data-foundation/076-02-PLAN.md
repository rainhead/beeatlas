```python
"""Canonical-name helper for Phase 76 / D-04.

Single source of truth for the 5-step name canonicalization rule used to
produce JOIN keys between checklist_data.species and ecdysis_data.occurrences.
"""

import re

# Step 1: strip authority. Two narrow patterns documented in RESEARCH.md Pitfall 1.
_AUTHORITY_RE = re.compile(r"\s*(?:,\s*\d{4}.*|\(\s*[A-ZÄÖÜÉÈ].*?\).*)\s*$")

# Step 2: strip subgenus parens — "(Initial-Cap-Word)" between binomial tokens.
_SUBGENUS_RE = re.compile(r"\s*\(\s*[A-Z][A-Za-zæ\-]+\s*\)\s*")

# Step 3: infraspecific markers — drop marker + everything after.
_INFRA_MARKERS = ("ssp.", "var.", "aff.", "cf.", "nr.", "subsp.")


def canonicalize(name: str | None) -> str | None:
    """Apply D-04 5-step canonicalization.

    Returns lowercase single-spaced binomial (or genus-only for higher-rank
    inputs), or None if input is None / empty / whitespace.

    Steps:
      1. Strip authority (drop ", <year>..." or " (<Author>...)..." tail)
      2. Strip subgenus parens ("Genus (Subgenus) species" -> "Genus species")
      3. Strip infraspecific markers (ssp./var./aff./cf./nr./subsp. + after);
         then fold trinomials to binomial (keep first 2 tokens)
      4. Lowercase
      5. Collapse whitespace to single space + trim
    """
    if name is None:
        return None
    s = name.strip()
    if not s:
        return None

    # Step 1: strip authority
    s = _AUTHORITY_RE.sub("", s)

    # Step 2: strip subgenus parens (replace with single space)
    s = _SUBGENUS_RE.sub(" ", s)

    # Step 3: strip infraspecific markers
    tokens = s.split()
    cleaned: list[str] = []
    for tok in tokens:
        if tok.lower() in _INFRA_MARKERS:
            break
        cleaned.append(tok)
    # Trinomial fold: keep at most first 2 tokens
    if len(cleaned) > 2:
        cleaned = cleaned[:2]

    # Steps 4 + 5: lowercase, collapse whitespace
    s = " ".join(t.lower() for t in cleaned)
    return s or None
```
