Rate-limit pattern from scripts/seed-species-photos.mjs: a `RateLimiter` class enforces a min-interval between calls (1000 ms documented; D-10 says "≤1 req/sec, matches seed-species-photos.mjs pacing"). The simplest portable approach is `await new Promise(r => setTimeout(r, 1000))` between requests — the script is single-threaded so no concurrency needed.

HEAD request: `fetch(url, { method: 'HEAD', redirect: 'follow' })` returns a Response with `.ok` and `.status`. Network errors throw — wrap in try/catch.

Drift report shape (concrete; consumed only by humans):
```json
{
  "checked_at": "2026-05-04T12:00:00.000Z",
  "total": 1424,
  "failures": [
    { "scientificName": "...", "photo_id": ..., "url": "...", "status": 404, "reason": "HTTP 404" },
    { "scientificName": "...", "photo_id": ..., "url": "...", "status": null, "reason": "fetch failed: ..." }
  ]
}
```

Workflow concurrency idiom (from D-10): top-level `concurrency: { group: photo-availability, cancel-in-progress: false }` — overlapping runs are queued or skipped, not cancelled (since each run is informational and skipping is fine).

```js
#!/usr/bin/env node
/**
 * PERF-04 (D-10): HEAD every photo URL in content/species-photos.toml,
 * write failures to data/manifest_drift_report.json. Informational only —
 * exits 0 even when failures exist. NOT in any build chain (mirrors the
 * PHOTO-07 isolation invariant from Phase 79).
 *
 * Pacing: <=1 req/sec (matches scripts/seed-species-photos.mjs).
 * Retry: single retry on HTTP >=500 with 2s backoff (D-10).
 */

import { readFileSync, writeFileSync, mkdirSync, existsSync } from 'node:fs';
import { dirname, join } from 'node:path';
import { fileURLToPath } from 'node:url';
import TOML from '@iarna/toml';

const here = dirname(fileURLToPath(import.meta.url));
const repoRoot = join(here, '..');
const TOML_PATH = join(repoRoot, 'content/species-photos.toml');
const REPORT_PATH = join(repoRoot, 'data/manifest_drift_report.json');
const RATE_MS = 1000;
const RETRY_BACKOFF_MS = 2000;

function sleep(ms) { return new Promise(r => setTimeout(r, ms)); }

async function headOnce(url) {
  try {
    const res = await fetch(url, { method: 'HEAD', redirect: 'follow' });
    return { ok: res.ok, status: res.status, reason: res.ok ? null : `HTTP ${res.status}` };
  } catch (e) {
    return { ok: false, status: null, reason: `fetch failed: ${e.message}` };
  }
}

async function checkUrl(url) {
  let result = await headOnce(url);
  // D-10: single retry on HTTP >=500 with 2s backoff. Network errors also retried once.
  const shouldRetry = !result.ok && (result.status === null || result.status >= 500);
  if (shouldRetry) {
    await sleep(RETRY_BACKOFF_MS);
    result = await headOnce(url);
  }
  return result;
}

function loadPhotos() {
  const manifest = TOML.parse(readFileSync(TOML_PATH, 'utf8'));
  const speciesTable = manifest.species ?? {};
  const out = [];
  for (const [name, entry] of Object.entries(speciesTable)) {
    for (const p of entry.photos ?? []) {
      if (typeof p.url === 'string' && p.url) {
        out.push({ scientificName: name, photo_id: p.photo_id, url: p.url });
      }
    }
  }
  return out;
}

async function main() {
  const photos = loadPhotos();
  const failures = [];
  let i = 0;
  for (const p of photos) {
    if (i > 0) await sleep(RATE_MS);
    i++;
    const r = await checkUrl(p.url);
    if (!r.ok) {
      failures.push({
        scientificName: p.scientificName,
        photo_id: p.photo_id,
        url: p.url,
        status: r.status,
        reason: r.reason,
      });
      console.warn(`! ${p.scientificName} ${p.photo_id}: ${r.reason}`);
    }
    if (i % 50 === 0) {
      console.log(`  progress: ${i}/${photos.length} (${failures.length} failures so far)`);
    }
  }
  const report = {
    checked_at: new Date().toISOString(),
    total: photos.length,
    failures,
  };
  if (!existsSync(dirname(REPORT_PATH))) mkdirSync(dirname(REPORT_PATH), { recursive: true });
  writeFileSync(REPORT_PATH, JSON.stringify(report, null, 2) + '\n');
  console.log(`Wrote ${REPORT_PATH}: ${failures.length}/${photos.length} failures`);
  // D-10: report-only. Exit 0 even when failures exist.
  process.exit(0);
}

if (fileURLToPath(import.meta.url) === (await import('node:path')).resolve(process.argv[1])) {
  main().catch(e => { console.error(`fatal: ${e.message}`); process.exit(0); /* still report-only */ });
}
```

Notes:
- Pacing uses `await sleep(RATE_MS)` BEFORE each request after the first (so total RPS is ~1).
- The catch block in `main()` exits 0 — D-10 says report-only / never blocks. A fatal exception (e.g. TOML parse failure) is logged but does not fail the workflow. This is intentional; the workflow's concurrency / cron cadence means a single bad run is self-recovering.
- No `package.json` script entry. The cron invokes the script by path. Adding an `npm run` wrapper would tempt future devs to chain it into `build` — DO NOT add one.

```yaml
# PERF-04 (D-10): weekly HEAD-check of every URL in content/species-photos.toml.
# Writes data/manifest_drift_report.json; commits the report if non-empty.
# Informational only — never blocks deploys (deploy.yml is a sibling workflow
# triggered by push, this one runs on schedule + manual dispatch only).
name: photo-availability

on:
  schedule:
    # Weekly, Mondays 13:00 UTC (early-morning Pacific). Adjust if it
    # collides with deploy.yml — they are independent but share runner quota.
    - cron: '0 13 * * 1'
  workflow_dispatch: {}

# D-10: skip overlapping runs rather than cancel — each run is informational.
concurrency:
  group: photo-availability
  cancel-in-progress: false

permissions:
  contents: write  # required to commit data/manifest_drift_report.json

jobs:
  check:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version-file: '.nvmrc'

      - name: Install dependencies
        run: npm ci

      - name: HEAD-check photo manifest
        run: node scripts/check-photo-availability.mjs

      - name: Commit drift report if changed
        run: |
          if [ -n "$(git status --porcelain data/manifest_drift_report.json)" ]; then
            git config user.name 'github-actions[bot]'
            git config user.email '41898282+github-actions[bot]@users.noreply.github.com'
            git add data/manifest_drift_report.json
            git commit -m "chore(82): refresh manifest_drift_report.json [skip ci]"
            git push
          else
            echo 'No drift change to commit.'
          fi
```

- The cron expression is documented inline. Adjust later if deploy.yml collisions surface.
