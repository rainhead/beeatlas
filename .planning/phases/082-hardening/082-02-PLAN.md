 * intrinsic size is honored once the image loads.
 */
bee-species-page bee-species-card > img {
  display: block;
  max-width: 100%;
  height: auto;
}

/* First <img> child of a card is the photo hero (per _pages/species.njk).

  aspect-ratio: 4 / 3;
  max-height: 360px;
  width: 100%;
  object-fit: contain;
  background: #f5f5f5;
}

/* SVG occurrence map: viewBox 600x320 = 15:8 aspect (MAP-02). */
bee-species-page bee-species-card > img[src*="/species-maps/"] {
  aspect-ratio: 15 / 8;
  width: 100%;
  max-width: 600px;
}

bee-species-page bee-species-card .attribution {

    grid-area: cards;
