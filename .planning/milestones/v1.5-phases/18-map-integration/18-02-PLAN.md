### 2. Add sample dot ghosting to `style.ts`

The `sampleDotStyle` function currently returns a color-by-recency style with no filter check. Add a ghost check BEFORE the cache lookup so filtered-out dots render as translucent grey:

```typescript
import { filterState, isFilterActive, matchesFilter } from './filter.ts';
// (import already present for clusterStyle — confirm it covers sampleDotStyle too)

export function sampleDotStyle(feature: FeatureLike): Style {
  // Ghost check — must come before cache lookup (ghost state depends on filterState)
  const active = isFilterActive(filterState);
  if (active && !matchesFilter(feature as Feature, filterState)) {
    // Return a single shared ghosted style — do not cache by tier
    return GHOSTED_SAMPLE_STYLE;
  }

  // ... rest of existing recency logic unchanged ...
}

// Add near top of style.ts (after imports):
const GHOSTED_SAMPLE_STYLE = new Style({
  image: new Circle({
    radius: 5,
    fill: new Fill({ color: 'rgba(170, 170, 170, 0.2)' }),
    stroke: new Stroke({ color: 'rgba(255,255,255,0.2)', width: 1 }),
  }),
});
```

Note: `sampleStyleCache` caches by recency tier only. The ghost check bypasses the cache — ghost state changes with filterState, not recency.

### 3. Add `_onPolygonClick` and `_clearRegionFilter` methods to `bee-map.ts`

```typescript
private _onPolygonClick(feature: Feature) {
  const isCounty = this.boundaryMode === 'counties';
  const name = isCounty
    ? (feature.get('NAME') as string)
    : (feature.get('NA_L3NAME') as string);
  const targetSet = isCounty ? filterState.selectedCounties : filterState.selectedEcoregions;
  const newSet = new Set(targetSet);
  if (newSet.has(name)) {
    newSet.delete(name);  // toggle deselect
  } else {
    newSet.add(name);
  }
  if (isCounty) {
    filterState.selectedCounties = newSet;
  } else {
    filterState.selectedEcoregions = newSet;
  }
  clusterSource.changed();
  sampleSource.changed();
  this.map?.render();
  // Update sidebar region text
  this._regionFilterText = this._buildRegionFilterText();
  // pushState per decision — each polygon click creates a browser history entry
  const view = this.map!.getView();
  const center = toLonLat(view.getCenter()!);
  const zoom = view.getZoom()!;
  const params = buildSearchParams(center, zoom, filterState, this._selectedOccIds, this.layerMode, this.boundaryMode);
  window.history.pushState({}, '', '?' + params.toString());
}

private _clearRegionFilter() {
  filterState.selectedCounties = new Set();
  filterState.selectedEcoregions = new Set();
  clusterSource.changed();
  sampleSource.changed();
  this.map?.render();
  this._regionFilterText = null;
  if (!this._isRestoringFromHistory && this.map) this._pushUrlState();
}

private _buildRegionFilterText(): string | null {
  const counties = [...filterState.selectedCounties];
  const ecors = [...filterState.selectedEcoregions];
  const all = [...counties, ...ecors];
  return all.length > 0 ? `Filter: ${all.join(', ')}` : null;
}
```

Add `@state() private _regionFilterText: string | null = null;` alongside the other @state() fields.

if (this.boundaryMode !== 'off') {
  const polyHits = await regionLayer.getFeatures(event.pixel);
  if (polyHits.length) {
    this._onPolygonClick(polyHits[0]! as Feature);
    return;

  // Miss on open map area — clear region filter

  return;
}

Critical ordering: polygon check must come AFTER the specimen/sample hit-test block. If `this.layerMode === 'specimens'` and specimen hits are found, the handler returns early before the polygon check. Same for samples. The polygon fallback only runs on a miss.
